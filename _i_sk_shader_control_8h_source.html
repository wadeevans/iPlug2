324  bool Rebuild()
325  {
326  auto [effect, errorText] = SkRuntimeEffect::Make(mShaderStr);
327  if (!effect) {
328  DBGMSG("%s\n", errorText.c_str());
329  return false;
330  }
331 
332  size_t oldSize = mRTEffect ? mRTEffect->inputSize() : 0;
333  mShaderInputs.realloc(effect->inputSize());
334  if (effect->inputSize() > oldSize) {
335  memset(mShaderInputs.get() + oldSize, 0, effect->inputSize() - oldSize);
336  }
337  mChildren.resize_back((int) effect->children().count());
338  for (auto& c : mChildren) {
339  if (!c) {
340  c = mShaders[0].second;
341  }
342  }
343 
344  mRTEffect = effect;
345 
346  auto inputs = SkData::MakeWithoutCopy(mShaderInputs.get(), mRTEffect->inputSize());
347  auto shader = mRTEffect->makeShader(std::move(inputs), mChildren.data(), mChildren.count(), nullptr, false);
348 
349  mPaint.setShader(std::move(shader));
350 
351  return true;
352  }
